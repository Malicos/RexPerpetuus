namespace = rp_abbasids
normal_or_historical_nations = yes

# End of the Seljuk State
country_event = {
	id = rp_abbasids.1
	title = rp_abbasids.1.t
	desc = rp_abbasids.1.d
	picture = RAMADAN_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = SJK
		has_regency = no
		has_consort_regency = no
		NOT = {
			dynasty = "Seljuk"
		}
	}
	
	# Become a regular tag
	option = {
		name = rp_abbasids.1.a
		ai_chance = { factor = 20 }
		if = {
			limit = {
				SJK = { any_owned_province = { province_id = 429 } }
				NOT = { exists = PER }
			}
			change_tag = PER
			change_primary_culture = persian
			swap_free_idea_group = yes
			429 = { move_capital_effect = yes }
			define_ruler = {
				ADM = 3
				DIP = 3
				MIL = 4
				fixed = yes
			}
		}
		else = {
			change_tag = IRQ
			change_primary_culture = al_iraqiya_arabic
			add_accepted_culture = turko_iranian
			swap_free_idea_group = yes
			410 = { move_capital_effect = yes }
			define_ruler = {
				ADM = 3
				DIP = 3
				MIL = 4
				fixed = yes
			}
		}
	}
	
	# The Caliph Takes Power
	option = {
		name = rp_abbasids.1.b
		ai_chance = { factor = 80 }
		country_event = { id = rp_abbasids.2 }
	}
}

		}
	}
}
