country_event = {                         #Start of event chain to cede english provinces in normandy and gascony in the HYW
	id = bt_harb_misc.6
	desc = bt_harb_misc.6.desc
	title = bt_harb_misc.6.title
	picture = CONQUEST_eventPicture
	fire_only_once = yes
	major = yes
	
	is_triggered_only = yes
	
	trigger = {	# Is france in HYW and occupies all english provinces in france except calais
		tag = FRA
	}
	
	# Cede all of Gascony and Normandy to France
	option = {
		name = bt_harb_misc.6.a
		ENG = {
			country_event = {
				id = bt_harb_misc.7
			}
		}
		custom_tooltip = bt_harb_misc.6.a.tt
		france_region = {
			limit = {
				has_province_flag = bt_hyw_concession
			}
			add_core = FRA
		}
		ai_chance = {
			factor = 4
		}
	}
	
	# Cede all of Gascony to France, but make Normandy into an Appanage
	option = {
		name = bt_harb_misc.6.b
		ENG = {
			country_event = {
				id = bt_harb_misc.7
			}
		}
		custom_tooltip = bt_harb_misc.6.b.tt
		hidden_effect = {
			france_region = {
				limit = {
					has_province_flag = bt_hyw_concession
				}
				if = {
					limit = {
						is_core = NRM
					}
					cede_province = NRM
				}
				else = {
					add_core = FRA
				}
			}
			NRM = {
				change_government = monarchy
				add_government_reform = feudalism_reform
				change_religion = ROOT
				change_primary_culture = normand
				set_prestige_to_zero = yes
				add_adm_power = 50
				add_dip_power = 50
				add_mil_power = 50
			}
		}
		tooltip = {
			release = NRM
		}
		create_subject = {
			subject_type = appanage
			subject = NRM
		}
		change_estate_land_share = {
			estate = estate_nobles
			share = -10
		}
		ai_chance = {
			factor = 1
		}
	}
	
	# Cede all of Normandy to France, but make Gascony into an Appanage
	option = {
		name = bt_harb_misc.6.c
		ENG = {
			country_event = {
				id = bt_harb_misc.7
			}
		}
		custom_tooltip = bt_harb_misc.6.c.tt
		hidden_effect = {
			france_region = {
				limit = {
					has_province_flag = bt_hyw_concession
				}
				if = {
					limit = { is_core = GUY }
					cede_province = GUY
				}
				else = {
					add_core = FRA
				}
			}
			GUY = {
				change_government = monarchy
				add_government_reform = feudalism_reform
				change_religion = ROOT
				change_primary_culture = gascon
				set_prestige_to_zero = yes
				add_adm_power = 50
				add_dip_power = 50
				add_mil_power = 50
			}
		}
		create_subject = {
			subject_type = appanage
			subject = GUY
		}
		change_estate_land_share = {
			estate = estate_nobles
			share = -2.5
		}
		ai_chance = {
			factor = 1
		}
	}
	
	# Make Gascony and Normandy into an Appanage
	option = {
		name = bt_harb_misc.6.d
		ENG = {
			country_event = {
				id = bt_harb_misc.7
			}
		}
		#custom_tooltip = bt_harb_misc.6.d.tt
		hidden_effect = {
			france_region = {
				limit = {
					has_province_flag = bt_hyw_concession
				}
				if = {
					limit = {
						is_core = GUY
					}
					cede_province = GUY
				}
				else_if = {
					limit = {
						is_core = NRM
					}
					cede_province = NRM
				}
				else = {
					add_core = FRA
				}
			}
			NRM = {
				change_government = monarchy
				add_government_reform = feudalism_reform
				change_religion = ROOT
				change_primary_culture = normand
				set_prestige_to_zero = yes
				add_adm_power = 50
				add_dip_power = 50
				add_mil_power = 50
			}
			GUY = {
				change_government = monarchy
				add_government_reform = feudalism_reform
				change_religion = ROOT
				change_primary_culture = gascon
				set_prestige_to_zero = yes
				add_adm_power = 50
				add_dip_power = 50
				add_mil_power = 50
			}
		}
		tooltip = {
			release = NRM
		}
		create_subject = {
			subject_type = appanage
			subject = NRM
		}
		tooltip = {
			release = GUY
		}
		create_subject = {
			subject_type = appanage
			subject = GUY
		}
		change_estate_land_share = {
			estate = estate_nobles
			share = -12.5
		}
		ai_chance = {
			factor = 2
		}
	}
	after = {
		every_province = {
			clr_province_flag = bt_hyw_concession
		}
	}
}
