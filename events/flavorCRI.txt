	immediate = {
		hidden_effect = {
			CRI = {
				every_subject_country = {
					limit = {
						is_subject_of_type = tributary_state
					}
					grant_independence = yes
				}
			}
			TUR = {
				if = {
					limit = {
						has_country_flag = tur_guarantee_cri_flag
					}
					create_march = CRI
				}
				else_if = {
					limit = {
						has_dlc = "Mandate of Heaven"
					}
					set_country_flag = forced_tributary_state
					create_subject = {
						subject_type = tributary_state
						subject = CRI
					}
					clr_country_flag = forced_tributary_state
				}
				else = { vassalize = CRI }
			}
		}
	}

	option = {
		name = "flavor_cri.4.a" #We have now tied our fate to theirs.
		add_legitimacy = 30
		add_horde_unity = 30

		add_mil_power = 100			#Up from 50
		add_army_professionalism = 0.05			#Added
		add_army_tradition = 5			#Added
		add_stability_or_adm_power = yes			#Some minor flavour changes, instead of adm
		#add_dip_power = 50			#Cut because I say so =P
		#add_adm_power = 50
		TUR = { create_march = CRI }
