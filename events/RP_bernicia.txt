namespace = bernicia

country_event = {
	id = bernicia.1
	title = bernicia.1.t
	desc = bernicia.1.d
	picture = RAMADAN_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = JRV
		any_province = {
			area = northern_england_area
			NOT = { owned_by = ROOT controlled_by = ROOT}
		}
	}

	mean_time_to_happen = { days = 1 }

	immediate = {
        set_global_flag = lost_bernicia
	}
	
	option = {
		name = bernicia.1.a
		add_country_modifier = {
			name = lost_bernicia_modifier
			duration = 36500
		}
		every_province = {
			limit = { area = northern_england_area }
			remove_core = ROOT
			remove_province_modifier = bernician_autonomy
            remove_province_modifier = shrinking_bernician_autonomy
		}
	}
}


country_event = {
	id = bernicia.2
	title = bernicia.2.t
	desc = bernicia.2.d
	picture = RAMADAN_eventPicture

	trigger = {
		tag = JRV
		any_owned_province = {
			OR = {
				has_province_modifier = bernician_autonomy
				has_province_modifier = shrinking_bernician_autonomy
			}
		}
		NOT = {
			has_country_modifier = bernicia_events_cooldown
		}
	}

	mean_time_to_happen = { days = 100 }

	immediate = {
		hidden_effect = {
			add_country_modifier = {
				name = bernicia_events_cooldown
				duration = 50
			}
			hidden = yes
		}
	}

	option = {
		name = bernicia.2.a
		random_owned_province = {
			limit = {
				OR = {
					has_province_modifier = bernician_autonomy
					has_province_modifier = shrinking_bernician_autonomy
				}
			}
			spawn_rebels = {
				type = noble_rebels
			}
			if = {
				limit = {
					has_province_modifier = bernician_autonomy
				}
				remove_province_modifier = bernician_autonomy
				add_province_modifier = {
					name = shrinking_bernician_autonomy
					duration = -1
				}
			}
			else = {
				remove_province_modifier = shrinking_bernician_autonomy
			}
			add_local_autonomy = -25
		}
		ai_chance = {
			factor = 90
		}
	}

	option = {
		name = bernicia.2.b
		add_years_of_income = -0.3
		ai_chance = {
			factor = 10
		}
	}
}
