aragonese_missions_1 = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		tag = ARA
		NOT = { has_dlc = "Golden Century" }
	}
	has_country_shield = yes

	vassalize_navarra = {
		icon = mission_have_two_subjects
		required_missions = { }
		provinces_to_highlight = {
			province_id = 210
			NOT = {
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		trigger = {
			210 = {
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		effect = {
			add_country_modifier = {
				name = military_vassalization
				duration = 7300
			}
			custom_tooltip = ara_crowns_of_iberia_claims
			hidden_effect = {
				every_province = {
					limit = {
						ara_crowns_of_iberia_provinces = yes
						NOT = { is_core = ROOT }
						NOT = { owned_by = ROOT }
					}
					add_permanent_claim = ROOT
				}
			}
		}
	}
	ara_crowns_of_iberia = {
		icon = mission_monarch_in_throne_room
		required_missions = { vassalize_navarra }
		provinces_to_highlight = {
			ara_crowns_of_iberia_provinces = yes
			NOT = {
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		trigger = {
			has_fulfiled_province_requirement_for_mission_trigger = {
				tooltip=country_holds_or_non_sovereign_subject_holds_highlighted 
				provinces=ara_crowns_of_iberia_provinces
				owner_type=country_or_non_sovereign_subject_holds
			}
		}
		effect = {
			add_adm_power = 100
			add_prestige = 20
			add_legitimacy = 20
			lower_andalucia_area = {
				limit = {
					NOT = { owned_by = ROOT	}
					NOT = { is_core = ROOT	}
				}
				add_claim = ROOT
			}
			upper_andalucia_area = {
				limit = {
					NOT = { owned_by = ROOT	}
					NOT = { is_core = ROOT	}
				}
				add_claim = ROOT
			}
		}
	}
}
