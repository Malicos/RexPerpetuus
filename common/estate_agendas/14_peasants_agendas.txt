#Development agendas
estate_peasants_develop_agricultural_tax_in_x = { #Copied from Clergy, modified
	selection_weight = {
		factor = 1
	}
	can_select = {
		any_owned_province = {
			OR = { #Farming goods. I exclude new-world farming goods because they were mostly tended to by slaves and the like. I also exclude alcohol-producing goods (Beer & Wine) because they were mainly tended to by monks.
				trade_goods = grain
				trade_goods = wool
				trade_goods = livestock
				trade_goods = bt_trade_good_fruit
			}
			is_state_core = root
		}
	}
	pre_effect = {
		random_list = {
			1 = {
				trigger = { #Added here because this can otherwise be selected when your capital area does not produce any of the below.
					any_owned_province = {
						is_state_core = root
						OR = { #Farming goods. I exclude new-world farming goods because they were mostly tended to by slaves and the like. I also exclude alcohol-producing goods (Beer & Wine) because they were mainly tended to by monks.
							trade_goods = grain
							trade_goods = wool
							trade_goods = livestock
							trade_goods = bt_trade_good_fruit
						}
						area_for_scope_province = {
							is_capital_of = root
						}
					}
				}
				random_owned_province = {
					limit = {
						is_state_core = root
						OR = { #Farming goods. I exclude new-world farming goods because they were mostly tended to by slaves and the like. I also exclude alcohol-producing goods (Beer & Wine) because they were mainly tended to by monks.
							trade_goods = grain
							trade_goods = wool
							trade_goods = livestock
							trade_goods = bt_trade_good_fruit
						}
						area_for_scope_province = {
							is_capital_of = root
						}
					}
					save_event_target_as = agenda_province
				}
			}
			2 = {
				trigger = {
					any_owned_province = {
						is_state_core = root
						OR = { #Farming goods. I exclude new-world farming goods because they were mostly tended to by slaves and the like. I also exclude alcohol-producing goods (Beer & Wine) because they were mainly tended to by monks.
							trade_goods = grain
							trade_goods = wool
							trade_goods = livestock
							trade_goods = bt_trade_good_fruit
						}
						region_for_scope_province = {
							is_capital_of = root
						}
					}
				}
				random_owned_province = {
					limit = {
						is_state_core = root
						OR = { #Farming goods. I exclude new-world farming goods because they were mostly tended to by slaves and the like. I also exclude alcohol-producing goods (Beer & Wine) because they were mainly tended to by monks.
							trade_goods = grain
							trade_goods = wool
							trade_goods = livestock
							trade_goods = bt_trade_good_fruit
						}
						OR = {
							AND = {
								religion = root
								OR = {
									culture = root
									has_owner_accepted_culture = yes
								}
							}
						}
						region_for_scope_province = {
							is_capital_of = root
						}
					}
					save_event_target_as = agenda_province
				}
			}
			1 = {
				random_owned_province = {
					limit = {
						is_state_core = root
						OR = { #Farming goods. I exclude new-world farming goods because they were mostly tended to by slaves and the like. I also exclude alcohol-producing goods (Beer & Wine) because they were mainly tended to by monks.
							trade_goods = grain
							trade_goods = wool
							trade_goods = livestock
							trade_goods = bt_trade_good_fruit
						}
					}
					save_event_target_as = agenda_province
				}
			}
		}
		event_target:agenda_province = {
			if = {
				limit = {
					check_variable = {
						which = province_adm_var
						value = 1
					}
				}
				set_variable = {
					which = province_adm_var
					value = 0
				}
			}
			export_to_variable = {
				which = province_adm_var
				value = trigger_value:base_tax
			}
			change_variable = {
				which = province_adm_var
				value = 2
			}
		}
	}
	provinces_to_highlight = {
		province_id = event_target:agenda_province
	}
	fail_if = {
		event_target:agenda_province = {
			NOT = { owned_by = root }
		}
	}
    task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_peasants
				}
			}
			has_estate_agenda_auto_completion = {
				estate = estate_peasants
			}
		}
		else = {
			event_target:agenda_province = {
				base_tax = "province_adm_var"
			}
		}
    }
    task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_peasants
		}
		add_estate_loyalty = {
			estate = estate_peasants
			loyalty = 10
		}
		add_prestige = 10
    }
    failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
		event_target:agenda_province = {
			if = {
				limit = { owned_by = root }
				add_named_unrest = {
					name = local_peasants_displeased
					value = 5
				}
			}
		}
        add_estate_loyalty_modifier = {
            estate = estate_peasants
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}

#Protect the countryside
estate_peasants_build_fort_building_in_y = { #Copied from Nobles, modified
	can_select = {
		any_owned_province = {
			OR = {
				AND = {
					NOT = { has_building = fort_15th }
					NOT = { has_building = fort_16th }
					NOT = { has_building = fort_17th }
					NOT = { has_building = fort_18th }
					num_free_building_slots = 1
					any_neighbor_province = {
						owner = {
							OR = {
								is_rival = root
								root = { is_rival = prev }
								AND = {
									is_subject_other_than_tributary_trigger = yes
									overlord = {
										OR = {
											is_rival = root
											root = { is_rival = prev }
										}
									}
								}
							}
						}
					}
					NOT = {
						any_neighbor_province = {
							owned_by = root
							OR = {
								has_building = fort_15th
								has_building = fort_16th
								has_building = fort_17th
								has_building = fort_18th
							}
						}
					}
				}
				AND = {
					NOT = { has_building = fort_15th }
					NOT = { has_building = fort_16th }
					NOT = { has_building = fort_17th }
