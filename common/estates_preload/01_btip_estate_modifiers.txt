estate_peasants = {
	modifier_definition = {
		type = loyalty
		key = peasants_loyalty_modifier
		trigger = {
			has_estate = estate_peasants
		}
	}
	
	modifier_definition = {
		type = loyalty
		key = peasants_absolutism_loyalty_modifier
		trigger = {
			has_estate = estate_peasants
			has_government_attribute = no_peasants_loyalty_loss_from_absolutism
		}
	}

	modifier_definition = {
		type = influence
		key = peasants_influence_modifier
		trigger = {
			has_estate = estate_peasants
		}
	}
	
	modifier_definition = {
		type = privileges
		key = peasants_privilege_slots
		trigger = {
			has_estate = estate_peasants
		}
	}
	#May potentially do something for peasant republic-specific modifiers. Will have to get back to this.
}

		trigger = {
			has_estate = estate_church

		}
	}
}
